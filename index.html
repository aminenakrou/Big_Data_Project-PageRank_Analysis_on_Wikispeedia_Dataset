pagerank_standard.py 
# Partie 1 : PageRank Standard
import numpy as np
import matplotlib.pyplot as plt
import time, os
from urllib.parse import unquote

def load_paths(file_path):
    if not os.path.exists(file_path):
        print(f"ERREUR : Fichier introuvable : {file_path}")
        return []
    with open(file_path, "r", encoding="utf-8") as f:
        lines = f.readlines()
    start_index = 0
    for i, line in enumerate(lines):
        if not line.startswith("#"):
            start_index = i; break
    paths = []
    for line in lines[start_index:]:
        parts = line.strip().split("\t")
        if len(parts) > 3:
            path_cleaned = [unquote(n) for n in parts[3].split(";")]
            paths.append(path_cleaned)
    print(f"Succès : {len(paths)} chemins chargés.")
    return paths

def create_node_index(paths):
    nodes = {}; index = 0
    for path in paths:
        for node in path:
            if node != "<" and node not in nodes:
                nodes[node] = index; index += 1
    return nodes

def matrice_transition(paths, nodes):
    n = len(nodes)
    matrice_adjacence = np.zeros((n, n), dtype=np.float32)
    for path in paths:
        visited = []
        for node in path:
            if node == "<":
                if len(visited) >= 2:
                    prev_i = nodes[visited[-1]]
                    curr_i = nodes[visited[-2]]
                    matrice_adjacence[prev_i, curr_i] += 1
                    visited.pop()
                elif visited: visited.pop()
            else:
                if visited:
                    prev_i = nodes[visited[-1]]
                    curr_i = nodes[node]
                    matrice_adjacence[prev_i, curr_i] += 1
                visited.append(node)
    matrice_trans = matrice_adjacence.T
    for j in range(n):
        col_sum = np.sum(matrice_trans[:, j])
        if col_sum > 0:
            matrice_trans[:, j] /= col_sum
        else:
            matrice_trans[:, j] = 1.0 / n
    return matrice_trans

def pagerank(matrix, beta, max_iterations=100, epsilon=1e-6):
    n = matrix.shape[0]
    pagerank_vec = np.ones(n) / n
    iteration = 0
    debut = time.time()
    teleport_vector = (1 - beta) / n
    while iteration < max_iterations:
        prev = pagerank_vec.copy()
        # PR = (1-β)/n + β × M × PR
        pagerank_vec = teleport_vector + beta * (matrix @ pagerank_vec)
        diff = np.linalg.norm(pagerank_vec - prev, ord=1)
        if diff < epsilon: break
        iteration += 1
    temps_exec = time.time() - debut
    return pagerank_vec, iteration, temps_exec
pagerank_personnalise.py 
# Partie 2 : PageRank Personnalisé
import numpy as np
import time, os, sys
from pagerank_standard import load_paths, create_node_index, matrice_transition

def pagerank_personnalise(matrix, beta, personal_nodes, nodes,
                            max_iterations=100, epsilon=1e-6):
    n = matrix.shape[0]
    pagerank = np.ones(n) / n
    personal_vector = np.zeros(n)

    # Construction du vecteur de personnalisation
    found_nodes = 0
    for node in personal_nodes:
        if node in nodes:
            personal_vector[nodes[node]] = 1
            found_nodes += 1
    if found_nodes == 0:
        personal_vector = np.ones(n) / n
    else:
        personal_vector /= np.sum(personal_vector)

    iteration = 0
    debut = time.time()
    diff = 1
    M_hat = matrix.T

    while iteration < max_iterations and diff >= epsilon:
        previous = pagerank.copy()
        # PR = (1-β) × V + β × Mᵀ × PR
        pagerank = (1 - beta) * personal_vector + beta * (M_hat @ previous)
        s = np.sum(pagerank)
        if s > 0: pagerank /= s
        diff = np.linalg.norm(pagerank - previous, ord=1)
        iteration += 1

    temps_exec = time.time() - debut
    return pagerank, iteration, temps_exec

def afficher_top_k(pagerank_vector, nodes_dict, k=20, titre="Résultats"):
    idx_to_node = {v: k for k, v in nodes_dict.items()}
    top_indices = np.argsort(pagerank_vector)[-k:][::-1]
    print(f"\n=== Top {k} : {titre} ===")
    for i, idx in enumerate(top_indices):
        nom = idx_to_node.get(idx, "Inconnu")
        score = pagerank_vector[idx]
        print(f"{i+1:<6} {nom:<35} {score:.6f}")
06 — Conclusion
Bilan & Perspectives
✅

Convergence efficace

Le PageRank standard avec β=0.85 converge en 26 itérations, validant l'approche par puissance itérée sur ce graphe réel.

🎯

Personnalisation pertinente

La variante personnalisée réoriente efficacement le classement vers des communautés thématiques cohérentes (informatique, faune africaine).

📈

Impact de β mesuré

Le facteur d'amortissement n'affecte pas le nœud dominant mais modifie significativement la hiérarchie des nœuds secondaires.

🚀

Perspectives

Applications potentielles en recommandation de contenu, analyse de réseaux sociaux et personnalisation de la navigation web.
